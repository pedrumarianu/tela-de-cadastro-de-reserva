        // JavaScript para listar as vagas (simulação)
        const vagasCadastradas = [
            { placa: "ABC1234", nome: "João", apartamento: "101", bloco: "A", modelo: "Sedan", cor: "Prata", vaga: "A101" },
            { placa: "XYZ5678", nome: "Maria", apartamento: "202", bloco: "B", modelo: "SUV", cor: "Preto", vaga: "B202" },
        ];

        const vagasList = document.getElementById("vagas-list");
        vagasCadastradas.forEach((vaga) => {
            const listItem = document.createElement("li");
            listItem.textContent = `Placa: ${vaga.placa}, Nome: ${vaga.nome}, Vaga: ${vaga.vaga}`;
            vagasList.appendChild(listItem);
        });
    </script>
